#include <iostream>
#include <math.h>
using namespace std;
int main() {
    float x,y,a,b,s;
    while (true)
    {
        cout<<"\nx="; cin>>x;
        cout<<"\ny="; cin>>y;
        a = pow(((x+1)/(x-1)),x);
        b = 18*x*pow(y,2);
        s = a+b;
        cout<<"\n Результут : "<<s;
    }

    return 0;
}
